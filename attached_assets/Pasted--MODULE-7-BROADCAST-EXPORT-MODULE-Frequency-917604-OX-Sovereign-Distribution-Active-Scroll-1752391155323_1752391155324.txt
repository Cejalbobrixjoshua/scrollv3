const { toPDF, toTXT, toMP3, toJSON } = require("./exportUtils");

function exportOutput(content, format, userId) {
  const fileName = `scroll_output_${userId}_${Date.now()}.${format}`;
  switch(format) {
    case 'pdf': return toPDF(content, fileName);
    case 'txt': return toTXT(content, fileName);
    case 'mp3': return toMP3(content, fileName);
    case 'json': return toJSON(content, fileName);
    default: throw new Error("Unsupported format");
  }
}
module.exports = { exportOutput };
